import tkinter
window = tkinter.Tk()
width = 600
height = 400
screen_width = window.winfo_screenwidth()
screen_height = window.winfo_screenheight()
x = screen_width/2-width/2
y = screen_height/2-height/2
window.geometry("%dx%d+%d+%d"%(width,height,x,y))


entry1 = tkinter.Entry(window,font=("宋体",27))
entry2 = tkinter.Entry(window,font=("宋体",27))
label = tkinter.Label(window,font=("宋体",20),bg="#FEF3F3",width=22,text="结果是：")


def result_add():
    label.config(text=int(entry1.get())+int(entry2.get()))
def result_jian():
    label.config(text=int(entry1.get())-int(entry2.get()))
def result_cheng():
    label.config(text=int(entry1.get())*int(entry2.get()))
def result_chu():
    label.config(text=int(entry1.get())/int(entry2.get()))

button1 = tkinter.Button(window,text="+",font=20,fg="#950FFF",command=result_add)
button2 = tkinter.Button(window,text="-",font=20,fg="#950FFF",command=result_jian)
button3 = tkinter.Button(window,text="*",font=20,fg="#950FFF",command=result_cheng)
button4 = tkinter.Button(window,text="/",font=20,fg="#950FFF",command=result_chu)
entry1.pack()
entry2.pack()
label.pack()
button1.pack(side="left",padx = 45,ipadx = 15,ipady = 15)
button2.pack(side="left",padx = 45,ipadx = 15,ipady = 15)
button3.pack(side="left",padx = 45,ipadx = 15,ipady = 15)
button4.pack(side="left",padx = 45,ipadx = 15,ipady = 15)

window.mainloop()
